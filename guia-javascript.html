Passo 01

Variáveis e Tipos de Dados

Variáveis são caixas nomeadas na memória que guardam valores. Em JavaScript moderno usamos let e const. Evite var — ele tem escopo confuso.

  • const — valor que não muda (constante)
  • let — valor que pode ser reatribuído
  • Tipos: number, string, boolean, null, undefined
JavaScript 
// ── Declarando variáveis ──────────────────
const nome   = "Maria";        // string (texto)
const idade  = 17;             // number (inteiro)
const nota   = 9.5;            // number (decimal)
const ativo  = true;           // boolean

let contador = 0;             // pode mudar depois
contador = 1;                // reatribuição OK

// ── Exibindo valores ──────────────────────
console.log("Nome:", nome);
console.log("Idade:", idade);
console.log("Nota:", nota);
console.log("Ativo?", ativo);
console.log("Tipo de nome:", typeof nome);
console.log("Tipo de nota:", typeof nota);
Saída
⚠ Atenção: JavaScript não exige declarar o tipo. O tipo é inferido pelo valor. Isso é chamado de tipagem dinâmica.
constlettypeof stringnumberboolean
Passo 02

Operadores

Operadores realizam cálculos, comparações e operações lógicas entre valores.

  • Aritméticos: + − * / % **
  • Comparação: === !== > < >= <=
  • Lógicos: && || !
  • Atribuição: = += -= *=
JavaScript 
const a = 10, b = 3;

// ── Aritméticos ───────────────────────────
console.log("Soma:",    a + b);   // 13
console.log("Subtração:", a - b); // 7
console.log("Divisão:",  a / b);  // 3.333...
console.log("Resto:",    a % b);  // 1
console.log("Potência:", a ** b); // 1000

// ── Comparação (sempre use ===) ───────────
console.log("a === 10?", a === 10); // true
console.log("a > b?",   a > b);   // true

// ── Lógicos ───────────────────────────────
const aprovado = a > 5 && b < 10;
console.log("Aprovado?", aprovado);  // true

// ── Template string (interpolação) ───────
console.log(`${a} dividido por ${b} = ${(a/b).toFixed(2)}`);
Saída
Use sempre === (igualdade estrita) em vez de ==, para evitar conversões de tipo inesperadas.
Passo 03

Estruturas Condicionais

Permitem que o programa tome decisões com base em condições verdadeiras ou falsas.

JavaScript 
// ── if / else if / else ───────────────────
const media = 6.8;

if (media >= 7) {
  console.log("✅ Aprovado");
} else if (media >= 5) {
  console.log("⚠️ Recuperação");
} else {
  console.log("❌ Reprovado");
}

// ── Operador ternário (forma curta) ───────
const status = media >= 7 ? "Aprovado" : "Reprovado";
console.log("Status:", status);

// ── switch (múltiplos casos fixos) ────────
const dia = 2;
switch (dia) {
  case 1: console.log("Segunda-feira"); break;
  case 2: console.log("Terça-feira");   break;
  case 3: console.log("Quarta-feira");  break;
  default: console.log("Outro dia");
}
Saída
Passo 04

Laços de Repetição

Laços executam um bloco de código várias vezes, evitando repetição manual.

  • for — quando sabe o número de repetições
  • while — enquanto uma condição for verdadeira
  • do...while — executa pelo menos uma vez
JavaScript 
// ── for clássico ─────────────────────────
console.log("=== Tabuada do 7 ===");
for (let i = 1; i <= 5; i++) {
  console.log(`7 × ${i} = ${7 * i}`);
}

// ── while ─────────────────────────────────
console.log("=== Contagem regressiva ===");
let n = 5;
while (n > 0) {
  console.log(n);
  n--;
}
console.log("🚀 Lançamento!");

// ── Soma com for ──────────────────────────
let soma = 0;
for (let k = 1; k <= 10; k++) soma += k;
console.log("Soma 1 a 10:", soma);
Saída
Passo 05

Vetores (Arrays)

Vetores armazenam múltiplos valores em sequência. O índice começa em 0.

  • Declaração: const v = [1, 2, 3]
  • Acesso: v[0], v[1]...
  • Métodos: push, pop, length, forEach, map, filter
JavaScript 
// ── Criando e acessando ───────────────────
const notas = [8.5, 7.0, 9.2, 6.8, 8.0];
console.log("Primeira nota:", notas[0]);
console.log("Qtd de notas:",  notas.length);

// ── Percorrendo com for ───────────────────
console.log("=== Todas as notas ===");
for (let i = 0; i < notas.length; i++) {
  console.log(`Aluno ${i+1}: ${notas[i]}`);
}

// ── Calculando a média ────────────────────
let total = 0;
notas.forEach(n => total += n);
const media = total / notas.length;
console.log("Média da turma:", media.toFixed(2));

// ── Filtrando aprovados ───────────────────
const aprovados = notas.filter(n => n >= 7);
console.log("Notas aprovadas:", aprovados);
Saída
Dica didática: Visualize o vetor como uma fila numerada, onde cada posição (índice) começa do zero.
Passo 06

Funções

Funções são blocos de código reutilizáveis que recebem parâmetros e retornam resultados. São o coração da programação modular.

  • Declaração clássica: function nome(params) {'{}'}
  • Arrow function: const nome = (params) => {'{}'}
  • Parâmetros com valor padrão
  • Retorno com return
JavaScript 
// ── Função clássica ───────────────────────
function saudar(nome) {
  return `Olá, ${nome}! Bem-vindo ao curso.`;
}
console.log(saudar("Carlos"));

// ── Função com múltiplos parâmetros ───────
function calcularMedia(...notas) {
  const soma = notas.reduce((acc, n) => acc + n, 0);
  return (soma / notas.length).toFixed(2);
}
console.log("Média:", calcularMedia(8, 7, 9, 6.5));

// ── Arrow function ────────────────────────
const dobro = x => x * 2;
console.log("Dobro de 7:", dobro(7));

// ── Parâmetro padrão ──────────────────────
function potencia(base, exp = 2) {
  return base ** exp;
}
console.log(potencia(4));       // 16 (exp=2 padrão)
console.log(potencia(2, 10));   // 1024

// ── Função que retorna vetor ───────────────
function filtrarAprovados(alunos) {
  return alunos.filter(a => a.nota >= 7);
}
const turma = [
  { nome: "Ana", nota: 8.5 },
  { nome: "Bob", nota: 5.0 },
  { nome: "Cia", nota: 9.0 },
];
const ok = filtrarAprovados(turma);
ok.forEach(a => console.log(`✅ ${a.nome}: ${a.nota}`));
Saída
Passo 07

Recursividade

Uma função recursiva é aquela que chama a si mesma. Toda recursão precisa de um caso base (condição de parada) para não entrar em loop infinito.

Estrutura mínima:
1) Caso base → retorna sem chamar a si mesma
2) Caso recursivo → resolve parte do problema e chama a si mesma
JavaScript 
// ── 1) Fatorial ───────────────────────────
// fat(5) = 5 × fat(4) = 5 × 4 × 3 × 2 × 1 = 120
function fatorial(n) {
  if (n <= 1) return 1;          // caso base
  return n * fatorial(n - 1);    // caso recursivo
}
console.log("5! =", fatorial(5));    // 120
console.log("7! =", fatorial(7));    // 5040

// ── 2) Fibonacci ──────────────────────────
// fib(n) = fib(n-1) + fib(n-2)
function fibonacci(n) {
  if (n <= 1) return n;            // caso base
  return fibonacci(n-1) + fibonacci(n-2);
}
console.log("Fibonacci(8):", fibonacci(8));  // 21

// ── 3) Soma de array com recursão ─────────
function somaArray(arr, i = 0) {
  if (i === arr.length) return 0;  // caso base
  return arr[i] + somaArray(arr, i + 1);
}
console.log("Soma [1..5]:", somaArray([1,2,3,4,5])); // 15

// ── 4) Contagem regressiva recursiva ──────
function contagem(n) {
  if (n === 0) { console.log("🚀"); return; }
  console.log(n + "...");
  contagem(n - 1);
}
contagem(4);
Saída
⚠ Atenção: Sempre defina o caso base primeiro. Sem ele, a função chamará a si mesma infinitamente até causar um "stack overflow" (estouro de pilha).
